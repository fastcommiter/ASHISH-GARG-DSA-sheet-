class Solution {
public:
    string mapWordWeights(vector<string>& words, vector<int>& weights) {
        string ans="";
        for(auto i :words)
            {
                int sum=0;
                for(char ch : i)
                    {
                        sum=sum+weights[ch-'a'];
                    }
                sum = sum % 26;
                ans.push_back('z' - sum);
            }
        return ans;
    }
};
